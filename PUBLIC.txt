<script>
// ── CONSTANTS ──────────────────────────────────────────────────────
const BACKEND = 'http://localhost:3001';
const REGISTRY = '0x56310d7e48d9249df358ab9daa6a2dad0e03e242';
const K1 = '0x01152d5c7467204bFa015061097b193CbceA8ca9';
const K2 = '0x55c73995c4194Dd87CC5aCbC4E45f48c807f4553';
const K3 = '0xA0eb06a5fab172860837C4D68e75F339896500b5';
const VEIL_PHRASE = 'Hopeology';
const VEIL_SALT = 'sgv1';
const RPC_URL = 'https://eth.llamarpc.com';

let provider;
let idleTimer, lockTimer;
const IDLE_WARN = 60, IDLE_LOCK = 120;
let operatorProof = '';

// ── VEIL ──────────────────────────────────────────────────────────
async function computeVeilHash(phrase) {
  const encoded = ethers.toUtf8Bytes(phrase + VEIL_SALT);
  return ethers.keccak256(encoded);
}

async function unlockVeil() {
  const phrase = document.getElementById('veilInput').value.trim();
  const expected = await computeVeilHash(VEIL_PHRASE);
  const given = await computeVeilHash(phrase);
  if (given === expected) {
    operatorProof = given;
    document.getElementById('veil').classList.add('hidden');
    document.getElementById('app').classList.remove('hidden');
    initApp();
    resetIdle();
  } else {
    document.getElementById('veilErr').textContent = 'Invalid operator phrase.';
    document.getElementById('veilInput').value = '';
  }
}
document.getElementById('veilInput').addEventListener('keydown', e => { if(e.key==='Enter') unlockVeil(); });

function hardLock() {
  document.getElementById('app').classList.add('hidden');
  document.getElementById('veil').classList.remove('hidden');
  document.getElementById('veilInput').value = '';
  document.getElementById('veilErr').textContent = '';
  operatorProof = '';
  clearInterval(idleTimer);
  document.getElementById('autoLockBar').style.display = 'none';
}

function resetIdle() {
  clearTimeout(lockTimer);
  document.getElementById('autoLockBar').style.display = 'none';
  let countdown = IDLE_WARN;
  clearInterval(idleTimer);
  idleTimer = setInterval(() => {
    countdown--;
    document.getElementById('lockCountdown').textContent = countdown;
    if (countdown <= IDLE_WARN) document.getElementById('autoLockBar').style.display = 'block';
    if (countdown <= 0) { clearInterval(idleTimer); hardLock(); }
  }, 1000);
}
['click','keydown','touchstart','mousemove'].forEach(e => document.addEventListener(e, resetIdle, {passive:true}));

// ── TABS ──────────────────────────────────────────────────────────
function switchTab(i) {
  document.querySelectorAll('.panel').forEach((p,idx) => p.classList.toggle('active', idx===i));
  document.querySelectorAll('.tab').forEach((t,idx) => t.classList.toggle('active', idx===i));
  if (i===1) pollRelays();
  if (i===2) pollBeacon();
}

// ── API HELPER ────────────────────────────────────────────────────
async function apiFetch(path, opts={}) {
  const headers = { 'Content-Type':'application/json', 'X-Operator-Proof': operatorProof, ...(opts.headers||{}) };
  const res = await fetch(BACKEND + path, { ...opts, headers });
  if (!res.ok) throw new Error(`API ${path} → ${res.status}`);
  return res.json();
}

// ── INIT ──────────────────────────────────────────────────────────
async function initApp() {
  try {
    provider = new ethers.JsonRpcProvider(RPC_URL);
    const network = await provider.getNetwork();
    document.getElementById('t-chain').textContent = `${network.name} (${network.chainId})`;
    document.getElementById('t-gate').textContent = REGISTRY;
    document.getElementById('t-k1').textContent = K1;
    document.getElementById('t-k2').textContent = K2;
    document.getElementById('t-k3').textContent = K3;
    pollTelemetry();
    setInterval(pollTelemetry, 15000);
    loadShareUrl();
  } catch(e) {
    document.getElementById('t-chain').textContent = 'RPC error: ' + e.message;
  }
}

async function pollTelemetry() {
  try {
    const [b1, b3] = await Promise.all([
      provider.getBalance(K1),
      provider.getBalance(K3)
    ]);
    document.getElementById('t-bal').textContent = ethers.formatEther(b1) + ' ETH';
    document.getElementById('t-bal3').textContent = ethers.formatEther(b3) + ' ETH';
    document.getElementById('t-poll').textContent = new Date().toLocaleTimeString();

    // Try backend for registry config
    try {
      const data = await apiFetch('/api/registry-state');
      document.getElementById('t-auth').textContent = data.authWindow ?? '—';
      document.getElementById('t-delay').textContent = data.minDelay ?? '—';
      appendEvent('Registry polled ✓');
    } catch { document.getElementById('t-auth').textContent = 'Backend offline'; }
  } catch(e) {
    document.getElementById('t-bal').textContent = 'RPC error';
  }
}

function appendEvent(msg) {
  const feed = document.getElementById('eventFeed');
  const ts = new Date().toLocaleTimeString();
  const el = document.createElement('div');
  el.style.cssText = 'padding:.25rem 0;border-bottom:1px solid var(--zinc800);color:var(--zinc400)';
  el.textContent = `[${ts}] ${msg}`;
  feed.prepend(el);
  if (feed.children.length > 20) feed.removeChild(feed.lastChild);
}

// ── RELAY MESH ────────────────────────────────────────────────────
async function pollRelays() {
  const grid = document.getElementById('relayGrid');
  grid.innerHTML = '<span style="color:var(--zinc500);font-size:.8rem">Polling…</span>';
  try {
    const data = await apiFetch('/api/relays');
    grid.innerHTML = '';
    data.relays.forEach(r => {
      const card = document.createElement('div');
      card.className = 'relay-card';
      const cls = r.status === 'ok' ? 'ok' : r.status === 'checking' ? 'checking' : 'fail';
      card.innerHTML = `<div class="relay-name">${r.name}</div><div class="relay-status ${cls}">${r.status.toUpperCase()}</div><div style="color:var(--zinc600);font-size:.7rem;margin-top:.2rem">${r.latency ?? ''}${r.latency ? 'ms' : ''}</div>`;
      grid.appendChild(card);
    });
    appendEvent(`Relay mesh polled — ${data.relays.filter(r=>r.status==='ok').length}/${data.relays.length} online`);
  } catch(e) {
    grid.innerHTML = `<span style="color:var(--red400);font-size:.8rem">Relay API error: ${e.message}</span>`;
  }
}

// ── BEACON ────────────────────────────────────────────────────────
async function pollBeacon() {
  document.getElementById('beaconStatus').textContent = 'checking…';
  try {
    const block = await provider.getBlock('latest');
    document.getElementById('beaconStatus').innerHTML = '<span class="status-dot dot-green"></span>Online';
    document.getElementById('beaconLast').textContent = new Date(block.timestamp * 1000).toLocaleString();
    document.getElementById('beaconBlock').textContent = '#' + block.number;
    appendEvent('Beacon pulse received block #' + block.number);
  } catch(e) {
    document.getElementById('beaconStatus').innerHTML = '<span class="status-dot dot-red"></span>Error';
  }
}

// ── EXPORT ────────────────────────────────────────────────────────
async function exportSource() {
  document.getElementById('exportStatus').textContent = 'Requesting export…';
  try {
    const data = await apiFetch('/api/export-source', { method: 'POST' });
    const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a'); a.href = url; a.download = 'securegate-v1-source-export.json'; a.click();
    URL.revokeObjectURL(url);
    document.getElementById('exportStatus').textContent = 'Export downloaded.';
    appendEvent('Source export downloaded');
  } catch(e) {
    document.getElementById('exportStatus').textContent = 'Export error: ' + e.message;
  }
}

// ── SHARE ─────────────────────────────────────────────────────────
async function loadShareUrl() {
  try {
    const data = await apiFetch('/api/share-url');
    document.getElementById('shareUrl').value = data.url || 'http://localhost:3001/';
  } catch { document.getElementById('shareUrl').value = 'http://localhost:3001/'; }
}

function copyShare() {
  const el = document.getElementById('shareUrl');
  el.select(); document.execCommand('copy');
  document.getElementById('shareMsg').textContent = 'Copied to clipboard!';
  setTimeout(() => document.getElementById('shareMsg').textContent = '', 2500);
}
